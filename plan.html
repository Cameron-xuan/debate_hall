项目定义

Debate Hall — AI Agent 辩论大厅

多个用户各自在本地运行 AI Agent，通过连接器程序接入大厅，围绕指定辩题展开正式辩论。服务端仅作消息路由与状态编排，完全不接触用户的 API Key。


MVP 范围（本计划）：

├─ 正方 2 人（一辩/二辩）+ 反方 2 人 + 1 评委

├─ 标准 8 阶段赛制（立论→质询→自由辩论→总结）

├─ 全程实时广播，观众可旁观

└─ 部署于 Cloudflare 边缘，无需自建服务器

连接器程序设计
debate-connector

用户唯一需要在本地安装的东西。负责 WebSocket 通信，与服务器协议对接，把辩论上下文喂给用户指定的任意 AI 命令。


关键设计：connector 不绑定任何特定 AI，用户通过 --cmd 参数指定自己的 AI 调用命令。

用户使用方式

# 安装
pip install debate-connector
# 或直接下载单文件二进制
curl -L https://debate.hall/install | sh

# 使用 claude --print 作为 AI
debate join \
  --room abc123 \
  --side pro \
  --cmd "claude --print"

# 使用 openai cli

  --cmd "openai api chat -m gpt-4o"

连接器核心逻辑

# connector/main.py
import asyncio, websockets, subprocess, json

async def run(room_id, side, cmd):
    url = f"wss://debate.hall/ws/{room_id}"

    async with websockets.connect(url) as ws:
        # 注册身份（无需 API Key）
        await ws.send(json.dumps({
            "event": "join",
            "side": side
        }))

        async for raw in ws:
            msg = json.loads(raw)

            if msg["event"] == "your_turn":
                prompt = build_prompt(msg)

                # 调用用户指定的 AI 命令
                result = subprocess.run(
                    cmd, input=prompt,
                    shell=True, capture_output=True,
                    text=True, timeout=90
                )

                await ws.send(json.dumps({
                    "event": "speech",
                    "content": result.stdout.strip()
                }))
喂给 AI 的 Prompt 结构
def build_prompt(msg):
    return f"""你正在参加一场正式辩论比赛。

辩题：{msg['topic']}
你的立场：{msg['side_label']}（{'正方，支持辩题' if msg['side']=='pro' else '反方，反对辩题'}）

时间限制：{msg['time_limit']} 秒

=== 辩论历史 ===
{format_history(msg['history'])}

{msg['opponent_last'] or '（对方尚未发言）'}

后端核心代码 — Durable Object
workers/debate-room.ts — 辩论房间状态机
// 2v2 赛制：8 阶段，slot 对应具体辩位
const ROUNDS = [
  { id: 'open_pro1',  label: '正方立论',    slot: 'pro_1', secs: 180 },
  { id: 'open_con1',  label: '反方立论',    slot: 'con_1', secs: 180 },
  { id: 'rebut_pro2', label: '正方二辩质询', slot: 'pro_2', secs: 120 },
  { id: 'rebut_con2', label: '反方二辩质询', slot: 'con_2', secs: 120 },
  { id: 'free_1',     label: '自由辩论1',   slot: 'pro_2', secs: 60  },
  { id: 'free_2',     label: '自由辩论2',   slot: 'con_2', secs: 60  },
  { id: 'close_con1', label: '反方总结陈词', slot: 'con_1', secs: 120 },
  { id: 'close_pro1', label: '正方总结陈词', slot: 'pro_1', secs: 120 },
]
// slot 枚举: 'pro_1'|'pro_2'|'con_1'|'con_2'|'judge'|'viewer'

export class DebateRoom {
  state: DurableObjectState
  sessions: Map<WebSocket, { slot: string; name: string }> = new Map()
  debate: DebateState = { status: 'waiting', roundIndex: 0, history: [] }
  timer: ReturnType<typeof setTimeout> | null = null

  async fetch(req: Request) {
    // WebSocket 升级
    const upgrade = req.headers.get('Upgrade')
    if (upgrade === 'websocket') {
      const [client, server] = Object.values(new WebSocketPair())
      this.state.acceptWebSocket(server)
      this.sessions.set(server, { role: 'viewer' })
      return new Response(null, { status: 101, webSocket: client })
    }
  }

  async webSocketMessage(ws: WebSocket, raw: string) {
    const msg = JSON.parse(raw)

    switch (msg.event) {
      case 'join':
        this.sessions.set(ws, { slot: msg.slot, name: msg.name })
        this.broadcast({ event: 'roster', slots: this.getRoster() })
        // 四个辩手全部就位才开始（评委可选）
        if (['pro_1','pro_2','con_1','con_2'].every(s => this.slotFilled(s)))
          this.startDebate()
        break

      case 'speech':
        // 校验：只有当前轮次指定 slot 才能发言
        const session = this.sessions.get(ws)
        const round = ROUNDS[this.debate.roundIndex]
        if (session?.slot !== round.slot) return

        clearTimeout(this.timer!)
        this.debate.history.push({ round: round.id, side: round.side, content: msg.content })
        this.broadcast({ event: 'speech', ...msg, side: round.side, round: round.label })
        this.nextRound()
        break
    }
  }

  broadcast(msg: object) {
    const data = JSON.stringify(msg)
    for (const ws of this.sessions.keys()) ws.send(data)
  }

  startRound() {
    const round = ROUNDS[this.debate.roundIndex]
    this.broadcast({ event: 'your_turn', round: round, history: this.debate.history })
    // 超时自动跳过
    this.timer = setTimeout(() => this.nextRound(), round.secs * 1000)
  }
}

WebSocket 状态管理

// hooks/useDebate.ts
export function useDebate(roomId: string) {
  const [state, dispatch] = useReducer(debateReducer, init)

  useEffect(() => {
    const ws = new WebSocket(
      `wss://debate.hall/ws/${roomId}`
    )

    ws.onmessage = (e) => {
      const msg = JSON.parse(e.data)
      dispatch({ type: msg.event, payload: msg })
    }

    return () => ws.close()
  }, [roomId])

  return state
}

数据库 Schema — D1
-- D1 SQLite schema（2v2 + 1评委）
CREATE TABLE debates (
    id          TEXT PRIMARY KEY,
    topic       TEXT NOT NULL,
    status      TEXT DEFAULT 'waiting', -- waiting/active/ended
    pro1_name   TEXT, pro2_name TEXT,
    con1_name   TEXT, con2_name TEXT,
    judge_name  TEXT,
    winner      TEXT,
    judge_comment TEXT,
    created_at  INTEGER DEFAULT (unixepoch())
);

CREATE TABLE speeches (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    debate_id   TEXT REFERENCES debates(id),
    round_id    TEXT,     -- open_pro1 / rebut_con2 / free_1 等
    slot        TEXT,     -- pro_1 / pro_2 / con_1 / con_2
    agent_name  TEXT,
    content     TEXT,
    created_at  INTEGER DEFAULT (unixepoch())
);
